# TCM-Flow V1.5 Public TCM-QG Evaluation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 以公开天池中医文献问题生成数据集 TCM-QG 作为论文主实验，验证结构化 Parent-Child 检索增强问答是否优于无证据回答、普通 chunk RAG 和 child-only 消融。

**Architecture:** 本计划只执行公开 TCM-QG 主实验，不再把自建 Formal-400 写入论文实验结果，也不执行自建 Parent-required stress subset。`train.json` 作为本地公开源文件输入，先冻结来源、许可证说明、过滤规则、doc-level split 和数据哈希；再从每个公开文档构建 parent document 与 child chunks，运行 `B0/B4/P/P-no-parent` 的检索与回答矩阵。若 TCM-QG 主实验达到预注册成功门槛，则不执行 MTCMB TCM-LitData；若未达到门槛，则停止并另写失败分析或后备公开小基准计划。

**Tech Stack:** Python 3.10、Pydantic 2、JSONL/CSV、PyYAML、NumPy、jieba BM25、BGE-M3、BGE Reranker、LangChain `ChatOpenAI`、OpenAI-compatible API、`unittest`

## 成功门槛

主实验成功必须同时满足：

```text
P_vs_B4_char_f1_delta > 0
P_vs_B4_char_f1_bootstrap_95ci_lower > 0
P_vs_P_no_parent_char_f1_delta > 0
P_vs_P_no_parent_char_f1_bootstrap_95ci_lower > 0
P_citation_recall >= P_no_parent_citation_recall
P_unsupported_answer_rate <= P_no_parent_unsupported_answer_rate
```

如果 `P vs B4` 成功但 `P vs P-no-parent` 不成功，论文只能写“结构化公开数据上的检索增强有效”，不能写“Parent context 带来稳定回答提升”。这种情况下仍视为 Parent-Child 主创新未完全成功，是否启用 MTCMB 需要用户重新确认。

PublicTcmQgSplit = Literal["train_pool", "dev", "test"]
PublicTcmQgMethod = Literal["B0", "B4", "P", "P-no-parent"]


class PublicTcmQgDocument(BaseModel):
    model_config = ConfigDict(extra="forbid")

    source_doc_id: str = Field(min_length=1)
    text: str = Field(min_length=1)
    annotations: list[dict[str, str]] = Field(min_length=1)

    @model_validator(mode="after")
    def validate_annotations(self) -> "PublicTcmQgDocument":
        for annotation in self.annotations:
            if not annotation.get("Q", "").strip():
                raise ValueError("annotation.Q 不能为空")
            if not annotation.get("A", "").strip():
                raise ValueError("annotation.A 不能为空")
        return self


class PublicTcmQgQaPair(BaseModel):
    model_config = ConfigDict(extra="forbid")

    qa_id: str = Field(min_length=1)
    source_doc_id: str = Field(min_length=1)
    split: PublicTcmQgSplit
    question: str = Field(min_length=1)
    answer: str = Field(min_length=1)
    source_text: str = Field(min_length=1)
    answer_start: int = Field(ge=0)
    answer_end: int = Field(gt=0)
    review_status: Literal["approved"]
    question_version: int = Field(default=1, ge=1)

    @model_validator(mode="after")
    def validate_answer_span(self) -> "PublicTcmQgQaPair":
        if self.answer_end <= self.answer_start:
            raise ValueError("answer_end 必须大于 answer_start")
        if self.source_text[self.answer_start : self.answer_end] != self.answer:
            raise ValueError("answer span 与 source_text 不一致")
        return self
```

import json
from datetime import datetime, timezone
from pathlib import Path

from experiments.rag_v1_5.runner import _atomic_write_json, _sha256_file


def _utc_now() -> str:
    return datetime.now(timezone.utc).isoformat().replace("+00:00", "Z")


def _read_tcm_qg_source(path: Path) -> list[dict]:
    return json.loads(path.read_text(encoding="utf-8-sig"))


def freeze_public_tcm_qg_source(
    *,
    source_path: Path,
    output_path: Path,
    public_dataset_url: str,
) -> dict:
    rows = _read_tcm_qg_source(source_path)
    qa_pair_count = sum(len(row.get("annotations", [])) for row in rows)
    manifest = {
        "version": "v1.5.0",
        "status": "ready",
        "stage": "public_tcm_qg_source_frozen",
        "generated_at": _utc_now(),
        "public_dataset_name": "Tianchi TCM-QG",
        "public_dataset_url": public_dataset_url,
        "source_file": source_path.name,
        "source_sha256": _sha256_file(source_path),
        "document_count": len(rows),
        "qa_pair_count": qa_pair_count,
        "privacy": {
            "raw_text_included": False,
            "full_source_committed": False,
        },
    }
    _atomic_write_json(output_path, manifest)
    return manifest
```

```python
import hashlib
import random

from experiments.rag_v1_5.schema import PublicTcmQgQaPair


def _normalize_text_key(value: str) -> str:
    return "".join(value.split())


def build_doc_split(
    *,
    doc_ids: list[str],
    seed: int,
    dev_rate: float,
    test_rate: float,
) -> dict[str, str]:
    ordered = sorted(set(doc_ids))
    rng = random.Random(seed)
    rng.shuffle(ordered)
    dev_count = round(len(ordered) * dev_rate)
    test_count = round(len(ordered) * test_rate)
    dev_ids = set(ordered[:dev_count])
    test_ids = set(ordered[dev_count : dev_count + test_count])
    return {
        doc_id: (
            "dev"
            if doc_id in dev_ids
            else "test"
            if doc_id in test_ids
            else "train_pool"
        )
        for doc_id in sorted(set(doc_ids))
    }


def normalize_public_tcm_qg_rows(
    *,
    rows: list[dict],
    split_by_doc_id: dict[str, str],
    min_text_chars: int,
    max_text_chars: int,
) -> list[dict]:
    normalized = []
    seen = set()
    for row in rows:
        source_doc_id = str(row["id"])
        source_text = row.get("text", "").strip().strip('"')
        if not (min_text_chars <= len(source_text) <= max_text_chars):
            continue
        split = split_by_doc_id[source_doc_id]
        for index, annotation in enumerate(row.get("annotations", [])):
            question = annotation.get("Q", "").strip()
            answer = annotation.get("A", "").strip()
            if not question or not answer:
                continue
            answer_start = source_text.find(answer)
            if answer_start < 0:
                continue
            dedupe_key = (
                source_doc_id,
                _normalize_text_key(question),
                _normalize_text_key(answer),
            )
            if dedupe_key in seen:
                continue
            seen.add(dedupe_key)
            answer_end = answer_start + len(answer)
            qa_hash = hashlib.sha1(
                f"{source_doc_id}:{index}:{question}:{answer}".encode("utf-8")
            ).hexdigest()[:12]
            payload = PublicTcmQgQaPair(

def _window_chunks(text: str, *, size: int, overlap: int) -> list[tuple[int, str]]:
    if size <= overlap:
        raise ValueError("chunk size 必须大于 overlap")
    chunks = []
    start = 0
    while start < len(text):
        piece = text[start : start + size]
        if piece.strip():
            chunks.append((start, piece))
        if start + size >= len(text):
            break
        start += size - overlap
    return chunks


def build_public_tcm_qg_chunks(
    *,
    rows: list[dict],
    child_chunk_size: int,
    child_chunk_overlap: int,
    b4_chunk_size: int,
    b4_chunk_overlap: int,
) -> list[dict]:
    docs = {}
    for row in rows:
        docs[row["source_doc_id"]] = row["source_text"]
    chunks = []
    for doc_id, text in sorted(docs.items()):
        parent_id = f"tcmqg-doc-{doc_id}"
        for strategy, size, overlap in (
            ("b4", b4_chunk_size, b4_chunk_overlap),
            ("child", child_chunk_size, child_chunk_overlap),
        ):
            for index, (start, piece) in enumerate(
                _window_chunks(text, size=size, overlap=overlap)
            ):
                chunks.append(
                    {
                        "chunk_id": f"tcmqg-{strategy}-{doc_id}-{index:04d}",
                        "source_doc_id": doc_id,
                        "strategy": strategy,
                        "text": piece,
                        "context_text": text if strategy == "child" else piece,
                        "retrieval_parent_id": parent_id,
                        "start_index": start,
                        "char_count": len(piece),
                    }
                )
    return chunks
```

```python
def _lcs_length(left: str, right: str) -> int:
    dp = [0] * (len(right) + 1)
    for left_char in left:
        previous = 0
        for index, right_char in enumerate(right, start=1):
            saved = dp[index]
            if left_char == right_char:
                dp[index] = previous + 1
            else:
                dp[index] = max(dp[index], dp[index - 1])
            previous = saved
    return dp[-1]


def rouge_l_f1(prediction: str, reference: str) -> float:
    if not prediction or not reference:
        return 0.0
    lcs = _lcs_length(prediction, reference)
    precision = lcs / len(prediction)
    recall = lcs / len(reference)
    if precision + recall == 0:
        return 0.0
    return 2 * precision * recall / (precision + recall)


def success_gate(*, comparisons: dict, by_method: dict) -> dict:
    p_b4 = comparisons["P-B4"]
    p_child = comparisons["P-P-no-parent"]
    success = (
        p_b4["char_f1_delta"] > 0
        and p_b4["char_f1_ci_low"] > 0
        and p_child["char_f1_delta"] > 0
        and p_child["char_f1_ci_low"] > 0
        and by_method["P"]["citation_recall"]
        >= by_method["P-no-parent"]["citation_recall"]
        and by_method["P"]["unsupported_answer_rate"]
        <= by_method["P-no-parent"]["unsupported_answer_rate"]
    )
    return {
        "public_tcm_qg_success": success,
        "skip_mtcmb_tcm_litdata": success,
    }
```
